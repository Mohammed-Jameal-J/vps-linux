VPS Hardening — Quick Recap

Unga VPS already purchased pannirukinga — idhu first 5 minutes la pannanum.

Idhu production system nu vechukinaal, namma andha basic security steps ah skip panna mudiyathu. Brief recap — full explanation venum na earlier beginner masterclass paarunga, but commands here irukku:

first 10 minutes on a fresh VPS
# 1. update everything
$ sudo apt update && sudo apt upgrade -y

# 2. create a non-root sudo user (never run production as root)
$ adduser deploy
$ usermod -aG sudo deploy

# 3. set up SSH key auth from your local machine
$ ssh-copy-id deploy@your_server_ip

# 4. disable password login + root login
$ sudo nano /etc/ssh/sshd_config
#    PasswordAuthentication no
#    PermitRootLogin no
$ sudo systemctl restart ssh

# 5. firewall — only what's needed
$ sudo ufw default deny incoming
$ sudo ufw allow OpenSSH
$ sudo ufw allow 80,443/tcp
$ sudo ufw enable

# 6. brute-force protection
$ sudo apt install fail2ban -y
$ sudo systemctl enable --now fail2ban
Idhu skip pannaadheenga

Production la real user data, payment info irukum — idhu test server alla. Andha 6 steps ah skip pannina, unga server first 24 hours kulla bots scan panni attack panna try pannum. 10 minutes investment, worth it.

  • Non-root sudo user created, root SSH login disabled
  • SSH key auth working, password auth disabled
  • UFW enabled — only 22, 80, 443 open
  • Fail2Ban running and watching SSH

    Realm

    Unga app ku ஒரு isolated space — users, roles, clients ellam ஒரு realm kulla irukும்.

    ⌘

    Client

    Unga React app / FastAPI backend — Keycloak kitta "register" aana application.

    ▣

    User

    ⎘

    FastAPI side — verify the JWT on every protected route

    backend/app/auth.py
    from fastapi import Depends, HTTPException
from fastapi.security import OAuth2AuthorizationCodeBearer
from jose import jwt
import httpx, os

KEYCLOAK_URL = os.environ.get("KEYCLOAK_URL")  # http://keycloak:8080
REALM = "myapp"

async def get_keycloak_public_key():
    async with httpx.AsyncClient() as client:
        resp = await client.get(f"{KEYCLOAK_URL}/realms/{REALM}")
        return resp.json()["public_key"]

async def verify_token(token: str = Depends(...)):
    try:
        public_key = await get_keycloak_public_key()
        payload = jwt.decode(
            token,
            f"-----BEGIN PUBLIC KEY-----\n{public_key}\n-----END PUBLIC KEY-----",
            algorithms=["RS256"],
            audience="account",
        )
        return payload  # contains user id, roles, email, etc.
    except Exception:
        raise HTTPException(status_code=401, detail="Invalid or expired token")

    frontend/src/Checkout.js
    import { useEffect, useState } from "react";

function Checkout({ token, orderId, amount }) {
  const [card, setCard] = useState(null);

  useEffect(() => {
    async function initSquare() {
      const payments = window.Square.payments(
        process.env.REACT_APP_SQUARE_APP_ID,
        process.env.REACT_APP_SQUARE_LOCATION_ID
      );
      const cardInstance = await payments.card();
      await cardInstance.attach("#card-container");
      setCard(cardInstance);
    }
    initSquare();
  }, []);

  async function handlePay() {
    const result = await card.tokenize();  // card never leaves the browser as raw data
    if (result.status === "OK") {
      const res = await fetch("/api/payments/charge", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({
          source_id: result.token,   // the one-time nonce
          order_id: orderId,
          amount_cents: amount,
        }),
      });
      const data = await res.json();
      console.log("Payment result:", data);
    }
  }

  return (
    <div>
      <div id="card-container"></div>
      <button onClick={handlePay}>Pay ${amount / 100}</button>
    </div>
  );
}

    from fastapi import APIRouter, Request, HTTPException
import hmac, hashlib, base64, os

router = APIRouter()
SIGNATURE_KEY = os.environ["SQUARE_WEBHOOK_SIGNATURE_KEY"]
NOTIFICATION_URL = "https://xyz.com/api/webhooks/square"

def verify_signature(body: bytes, signature: str) -> bool:
    # Square signs: HMAC-SHA256(notification_url + raw_body, signature_key)
    string_to_sign = NOTIFICATION_URL + body.decode()
    computed = hmac.new(
        SIGNATURE_KEY.encode(), string_to_sign.encode(), hashlib.sha256
    ).digest()
    computed_b64 = base64.b64encode(computed).decode()
    return hmac.compare_digest(computed_b64, signature)

@router.post("/api/webhooks/square")
async def square_webhook(request: Request):
    raw_body = await request.body()
    signature = request.headers.get("x-square-hmacsha256-signature", "")

    if not verify_signature(raw_body, signature):
        raise HTTPException(status_code=401, detail="Invalid webhook signature")

    payload = await request.json()
    event_type = payload.get("type")

    if event_type == "payment.updated":
        payment = payload["data"]["object"]["payment"]
        if payment["status"] == "COMPLETED":
            # mark order as paid in your own database here
            pass

    return {"received": True}
